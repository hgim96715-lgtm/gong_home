## 정답 쿼리 (Solution)

```sql
-- 여기에 작성한 쿼리를 붙여넣으세요.
WITH best AS (
  SELECT 
    author,
    year,
    year - ROW_NUMBER() OVER (PARTITION BY author ORDER BY year) AS yn
  FROM (
    SELECT DISTINCT 
      author, 
      year 
    FROM books 
    WHERE genre = 'Fiction'
  ) AS f
)
SELECT 
  author, 
  MAX(year) AS year, 
  COUNT(*) AS depth
FROM best
GROUP BY author, yn
HAVING COUNT(*) >= 5
ORDER BY depth DESC, author ASC;
```
